  <script>
    function changeContent(page) {
      const content = document.getElementById('content');
      switch(page) {
        case 'home':
          content.innerHTML = `
            <img src="https://media.geeksforgeeks.org/wp-content/uploads/geeksforgeeks-12.png" alt="Home">
            <h2>Home Page</h2>
            <p>Welcome to our SPA. Use the navigation bar to move through sections without reloading.</p>
          `;
          break;
        case 'about':
          content.innerHTML = `
            <h2>About This App</h2>
            <p>This is a lightweight SPA made using only HTML, CSS, and vanilla JavaScript.</p>
            <p>It demonstrates dynamic content rendering and simple routing concepts.</p>
          `;
          break;
        case 'contact':
          content.innerHTML = `
            <h2>Contact Us</h2>
            <form>
              <input type="text" placeholder="Your Name" required>
              <input type="email" placeholder="Your Email" required>
              <textarea placeholder="Your Message" rows="4"></textarea>
              <button type="submit">Send</button>
            </form>
          `;
          break;
        default:
          content.innerHTML = `<h2>404 - Page not found</h2>`;
      }
    }
  </script>
